# 前端设计模式

## 1. 观察者 / 发布订阅模式

```javascript
// 发布订阅（EventEmitter）
class EventBus {
  constructor() { this.events = new Map(); }

  on(event, callback) {
    if (!this.events.has(event)) this.events.set(event, []);
    this.events.get(event).push(callback);
    return () => this.off(event, callback); // 返回取消订阅函数
  }

  off(event, callback) {
    const callbacks = this.events.get(event);
    if (callbacks) {
      this.events.set(event, callbacks.filter(cb => cb !== callback));
    }
  }

  emit(event, ...args) {
    this.events.get(event)?.forEach(cb => cb(...args));
  }
}

// 使用
const bus = new EventBus();
const unsubscribe = bus.on('userLogin', (user) => console.log('登录:', user));
bus.emit('userLogin', { name: '张三' });
unsubscribe(); // 取消订阅
```

## 2. 策略模式

```javascript
// 表单验证策略
const validators = {
  required: (value) => value ? null : '此字段必填',
  email: (value) => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value) ? null : '邮箱格式不正确',
  minLength: (min) => (value) => value.length >= min ? null : `最少${min}个字符`,
  maxLength: (max) => (value) => value.length <= max ? null : `最多${max}个字符`,
};

function validate(value, rules) {
  for (const rule of rules) {
    const error = rule(value);
    if (error) return error;
  }
  return null;
}

// 使用
const error = validate('ab', [validators.required, validators.minLength(3)]);
// '最少3个字符'
```

## 3. 单例模式

```javascript
// 全局状态管理、日志、配置
class Logger {
  static instance;
  static getInstance() {
    if (!Logger.instance) Logger.instance = new Logger();
    return Logger.instance;
  }
  log(msg) { console.log(`[${new Date().toISOString()}] ${msg}`); }
}
```

function withAuth(Component) {
  return function AuthenticatedComponent(props) {
    const { user } = useAuth();
    if (!user) return <Navigate to="/login" />;
    return <Component {...props} user={user} />;
  };
}

// Render Props
function MouseTracker({ render }) {
  const [pos, setPos] = useState({ x: 0, y: 0 });
  useEffect(() => {
    const handler = (e) => setPos({ x: e.clientX, y: e.clientY });
    window.addEventListener('mousemove', handler);
    return () => window.removeEventListener('mousemove', handler);
  }, []);
  return render(pos);
}

## 5. MVC / MVP / MVVM

```
MVC：Model-View-Controller（传统 Web）
MVP：Model-View-Presenter（Android）
MVVM：Model-View-ViewModel（Vue、React）
  - View（组件模板）↔ ViewModel（组件逻辑/状态）↔ Model（数据/API）
  - 双向数据绑定（Vue）或单向数据流（React）
```
