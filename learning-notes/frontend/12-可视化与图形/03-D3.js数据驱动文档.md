## 1. 核心概念

```javascript
import * as d3 from 'd3';

// 选择集
d3.select('#chart');          // 选择单个元素
d3.selectAll('.bar');         // 选择所有匹配元素

// 数据绑定（Enter-Update-Exit 模式）
const data = [10, 20, 30, 40, 50];

const bars = d3.select('#chart')
  .selectAll('rect')
  .data(data);

// Enter：新数据，创建元素
bars.enter()
  .append('rect')
  .attr('x', (d, i) => i * 60)
  .attr('y', d => 300 - d * 5)
  .attr('width', 50)
  .attr('height', d => d * 5)
  .attr('fill', '#3498db');

// Update：已有数据，更新元素
bars.attr('height', d => d * 5);

// Exit：多余元素，移除
bars.exit().remove();

// 简写（join，D3 v6+）
d3.select('#chart')
  .selectAll('rect')
  .data(data)
  .join('rect')
  .attr('x', (d, i) => i * 60)
  .attr('y', d => 300 - d * 5)
  .attr('width', 50)
  .attr('height', d => d * 5)
  .attr('fill', '#3498db');
```

## 2. 比例尺

```javascript
// 线性比例尺
const xScale = d3.scaleLinear()
  .domain([0, 100])     // 数据范围
  .range([0, 800]);     // 像素范围

// 序数比例尺
const colorScale = d3.scaleOrdinal()
  .domain(['A', 'B', 'C'])
  .range(['#e74c3c', '#3498db', '#2ecc71']);

// 带状比例尺（柱状图）
const bandScale = d3.scaleBand()
  .domain(['1月', '2月', '3月'])
  .range([0, 400])
  .padding(0.2);

// 时间比例尺
const timeScale = d3.scaleTime()
  .domain([new Date(2024, 0, 1), new Date(2024, 11, 31)])
  .range([0, 800]);
```

## 3. 坐标轴

```javascript
const svg = d3.select('#chart').append('svg').attr('width', 800).attr('height', 400);
const margin = { top: 20, right: 20, bottom: 30, left: 40 };

const xAxis = d3.axisBottom(xScale);
const yAxis = d3.axisLeft(yScale);

svg.append('g')
  .attr('transform', `translate(0, ${400 - margin.bottom})`)
  .call(xAxis);

svg.append('g')
  .attr('transform', `translate(${margin.left}, 0)`)
  .call(yAxis);
```

## 4. 过渡动画

```javascript
d3.selectAll('rect')
  .transition()
  .duration(750)
  .ease(d3.easeCubicOut)
  .attr('height', d => yScale(d))
  .attr('fill', '#2ecc71');
```
