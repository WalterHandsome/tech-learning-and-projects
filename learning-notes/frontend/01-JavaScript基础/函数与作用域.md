## 1. 函数声明与表达式

```javascript
// 函数声明（会提升）
function greet(name) {
  return `Hello, ${name}`;
}

// 函数表达式（不会提升）
const greet = function(name) {
  return `Hello, ${name}`;
};

// 箭头函数（没有自己的 this、arguments、prototype）
const greet = (name) => `Hello, ${name}`;
const add = (a, b) => a + b;
const getObj = () => ({ name: '张三' }); // 返回对象需要括号

// 立即执行函数（IIFE）
(function() {
  console.log('立即执行');
})();
```

## 2. 作用域

```javascript
// 全局作用域
var globalVar = 'global';

// 函数作用域
function fn() {
  var localVar = 'local';
  console.log(globalVar); // 'global'
}
// console.log(localVar); // ReferenceError

// 块级作用域（let/const）
{
  let blockVar = 'block';
  const blockConst = 'const';
}
// console.log(blockVar); // ReferenceError
```

## 3. 闭包

```javascript
// 闭包：函数能够访问其词法作用域中的变量，即使函数在其作用域之外执行
function createCounter() {
  let count = 0;
  return {
    increment: () => ++count,
    decrement: () => --count,
    getCount: () => count,
  };
}

const counter = createCounter();
counter.increment(); // 1
counter.increment(); // 2
counter.getCount();  // 2

// 闭包经典问题
for (var i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 100); // 3, 3, 3
}

// 解决方案1：let
for (let i = 0; i < 3; i++) {
  setTimeout(() => console.log(i), 100); // 0, 1, 2
}

// 解决方案2：闭包
for (var i = 0; i < 3; i++) {
  ((j) => {
    setTimeout(() => console.log(j), 100);
  })(i); // 0, 1, 2
}
```

## 4. this 指向

```javascript
// 1. 默认绑定（非严格模式指向 window，严格模式 undefined）
function fn() { console.log(this); }
fn(); // window（非严格模式）

// 2. 隐式绑定（谁调用指向谁）
const obj = {
  name: '张三',
  greet() { console.log(this.name); }
};
obj.greet(); // '张三'

// 3. 显式绑定
function greet() { console.log(this.name); }
const user = { name: '李四' };
greet.call(user);    // '李四'（立即调用）
greet.apply(user);   // '李四'（立即调用，参数为数组）
const bound = greet.bind(user); // 返回新函数
bound(); // '李四'

// 4. new 绑定
function Person(name) {
  this.name = name;
}
const p = new Person('王五'); // this 指向新对象

// 5. 箭头函数（继承外层 this，不能被 call/apply/bind 改变）
const obj = {
  name: '张三',
  greet: () => {
    console.log(this.name); // undefined（继承外层，非 obj）
  },
  delayGreet() {
    setTimeout(() => {
      console.log(this.name); // '张三'（继承 delayGreet 的 this）
    }, 100);
  }
};
```

## 5. call / apply / bind

```javascript
// call：逐个传参
function introduce(age, city) {
  console.log(`${this.name}, ${age}岁, 来自${city}`);
}
introduce.call({ name: '张三' }, 25, '北京');

// apply：数组传参
introduce.apply({ name: '李四' }, [30, '上海']);

// bind：返回新函数
const boundFn = introduce.bind({ name: '王五' }, 28);
boundFn('广州');

// 手写 bind
Function.prototype.myBind = function(context, ...args) {
  const fn = this;
  return function(...newArgs) {
    return fn.apply(context, [...args, ...newArgs]);
  };
};
```
