# 微信小程序开发

## 1. 项目结构

```
├── app.js          # 小程序入口
├── app.json        # 全局配置
├── app.wxss        # 全局样式
├── pages/
│   ├── index/
│   │   ├── index.js    # 页面逻辑
│   │   ├── index.wxml  # 页面模板
│   │   ├── index.wxss  # 页面样式
│   │   └── index.json  # 页面配置
│   └── detail/
├── components/     # 自定义组件
└── utils/          # 工具函数
```

## 2. WXML 模板

```xml
<!-- 数据绑定 -->
<view>{{ message }}</view>

<!-- 条件渲染 -->
<view wx:if="{{ type === 1 }}">类型1</view>
<view wx:elif="{{ type === 2 }}">类型2</view>
<view wx:else>其他</view>

<!-- 列表渲染 -->
<view wx:for="{{ items }}" wx:key="id">
  {{ index }}: {{ item.name }}
</view>

<!-- 事件绑定 -->
<button bindtap="handleTap" data-id="{{ item.id }}">点击</button>
```

## 3. 页面逻辑

```javascript
Page({
  data: { message: 'Hello', items: [] },

  onLoad(options) { this.fetchData(); },
  onShow() {},
  onReady() {},
  onHide() {},
  onUnload() {},
  onPullDownRefresh() { this.fetchData(); wx.stopPullDownRefresh(); },
  onReachBottom() { this.loadMore(); },

  handleTap(e) {
    const id = e.currentTarget.dataset.id;
    wx.navigateTo({ url: `/pages/detail/detail?id=${id}` });
  },

  async fetchData() {
    const res = await wx.request({ url: 'https://api.example.com/data', method: 'GET' });
    this.setData({ items: res.data });
  },
});
```

## 4. 跨端框架

```javascript
// Taro（React 语法写小程序）
function Index() {
  const [list, setList] = useState([]);
  useEffect(() => { Taro.request({ url: '/api/list' }).then(res => setList(res.data)); }, []);
  return (
    <View>{list.map(item => <Text key={item.id}>{item.name}</Text>)}</View>
  );
}
```
