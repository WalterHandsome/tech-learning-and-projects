from sqlalchemy import event
from sqlalchemy.engine import Engine
import time

@event.listens_for(Engine, "before_cursor_execute")
def receive_before_cursor_execute(conn, cursor, statement, parameters, context, executemany):
    conn.info.setdefault('query_start_time', []).append(time.time())

@event.listens_for(Engine, "after_cursor_execute")
def receive_after_cursor_execute(conn, cursor, statement, parameters, context, executemany):
    total = time.time() - conn.info['query_start_time'].pop(-1)
    if total > 1.0:  # ж…ўжҹҘиҜўйҳҲеҖј
        logger.warning(f"ж…ўжҹҘиҜў: {statement[:100]} иҖ—ж—¶: {total:.2f}s")

## 7. жҖ»з»“

зӣ‘жҺ§дёҺж—Ҙеҝ—иҰҒзӮ№пјҡ
- **еә”з”Ёзӣ‘жҺ§**пјҡPrometheusгҖҒеҒҘеә·жЈҖжҹҘгҖҒжҢҮж Үж”¶йӣҶ
- **ж—Ҙеҝ—иҒҡеҗҲ**пјҡELK StackгҖҒз»“жһ„еҢ–ж—Ҙеҝ—
- **й”ҷиҜҜиҝҪиёӘ**пјҡSentryгҖҒиҮӘе®ҡд№үй”ҷиҜҜиҝҪиёӘ
- **жҖ§иғҪзӣ‘жҺ§**пјҡAPMгҖҒж…ўжҹҘиҜўзӣ‘жҺ§
- **иө„жәҗзӣ‘жҺ§**пјҡCPUгҖҒеҶ…еӯҳгҖҒзЈҒзӣҳзӣ‘жҺ§
- **е‘ҠиӯҰзі»з»ҹ**пјҡе‘ҠиӯҰи§„еҲҷгҖҒйҖҡзҹҘжңәеҲ¶
