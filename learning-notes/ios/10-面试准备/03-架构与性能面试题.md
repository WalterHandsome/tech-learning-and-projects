## 1. MVC vs MVVM

```swift
// MVC 问题：Massive View Controller
// Controller 承担：网络请求、数据处理、UI 更新、事件响应
// 导致：难以测试、代码臃肿、职责不清

// MVVM 优势：
// - ViewModel 可独立测试（不依赖 UIKit）
// - 数据绑定减少胶水代码
// - 职责清晰：View 只负责展示

class UserViewModel: ObservableObject {
    @Published var displayName = ""
    @Published var isLoading = false

    private let useCase: FetchUserUseCase

    init(useCase: FetchUserUseCase) { self.useCase = useCase }

    // 可测试的业务逻辑
    @MainActor
    func load(userId: Int) async {
        isLoading = true
        defer { isLoading = false }
        if let user = try? await useCase.execute(id: userId) {
            displayName = "\(user.name) (\(user.email))"
        }
    }
}

// 测试 ViewModel（无需 UI）
func testLoadUser() async {
    let mockUseCase = MockFetchUserUseCase()
    let vm = UserViewModel(useCase: mockUseCase)
    await vm.load(userId: 1)
    XCTAssertEqual(vm.displayName, "张三 (zhang@test.com)")
}
```

## 2. 启动优化策略

```swift
// 冷启动优化清单：
// Pre-main 阶段：
// 1. 减少动态库（合并为静态库）
// 2. 减少 ObjC 类数量
// 3. 避免 +load 方法

// Post-main 阶段：
// 1. didFinishLaunching 只做必要初始化
// 2. 延迟非首屏依赖的初始化
// 3. 首屏数据预加载 + 缓存

class AppLauncher {
    func launch() {
        // 阶段1：同步必要初始化
        setupCrashReporter()
        setupRootWindow()

        // 阶段2：首帧渲染后异步初始化
        DispatchQueue.main.async {
            self.setupAnalytics()
            self.setupPush()
            self.prefetchData()
        }

        // 阶段3：后台低优先级初始化
        DispatchQueue.global(qos: .utility).async {
            self.warmUpDatabase()
            self.cleanExpiredCache()
        }
    }
}
```

## 4. 列表性能优化

```swift
// 核心优化点：
// 1. Cell 复用 + prepareForReuse 重置状态
// 2. 避免离屏渲染（cornerRadius + clipsToBounds）
// 3. 异步图片加载 + 降采样
// 4. 高度缓存（避免重复计算）
// 5. 预加载（prefetchRowsAt）

class OptimizedCell: UITableViewCell {
    override func prepareForReuse() {
        super.prepareForReuse()
        imageView?.image = nil
        currentTask?.cancel()
    }

    func configure(with item: Item) {
        // 文本直接赋值（主线程）
        titleLabel.text = item.title

        // 图片异步加载 + 降采样
        currentTask = Task {
            let image = await ImageLoader.shared.load(
                url: item.imageURL,
                targetSize: CGSize(width: 60, height: 60)
            )
            guard !Task.isCancelled else { return }
            avatarView.image = image
        }
    }
}
```

## 5. 网络层设计

```swift
// 面试常问：如何设计一个可测试、可扩展的网络层

protocol APIEndpoint {
    var path: String { get }
    var method: HTTPMethod { get }
    var headers: [String: String] { get }
    associatedtype Response: Decodable
}

struct GetUsersEndpoint: APIEndpoint {
    typealias Response = [User]
    var path = "/users"
    var method: HTTPMethod = .get
    var headers: [String: String] = [:]
}

protocol NetworkClient {
    func send<E: APIEndpoint>(_ endpoint: E) async throws -> E.Response
}

// 真实实现
class URLSessionNetworkClient: NetworkClient {
    func send<E: APIEndpoint>(_ endpoint: E) async throws -> E.Response {
        let (data, _) = try await URLSession.shared.data(for: endpoint.urlRequest)
        return try JSONDecoder().decode(E.Response.self, from: data)
    }
}

// Mock 实现（测试用）
class MockNetworkClient: NetworkClient {
    var mockData: Data?
    func send<E: APIEndpoint>(_ endpoint: E) async throws -> E.Response {
        return try JSONDecoder().decode(E.Response.self, from: mockData!)
    }
}
```

## 7. 2026 年架构与性能面试新题

### 7.1 SwiftUI 架构面试题

```swift
// Q: 2026 年 iOS 项目推荐什么架构？
// A: SwiftUI + @Observable + Swift Concurrency 是主流：
// - @Observable 替代 @ObservableObject（更细粒度、更高性能）
// - async/await 替代 Combine（更简洁、更易理解）
// - Swift 6.2 默认 MainActor 隔离简化了线程安全
// - 仍然推荐 MVVM 或 TCA（The Composable Architecture）

// Q: @Observable 和 @ObservableObject 的性能差异？
// A: @Observable 性能更好：
// - 属性级别观察：只有被读取的属性变化才触发视图更新
// - @ObservableObject 是对象级别：任何 @Published 属性变化都触发更新
// - 在复杂视图层级中，@Observable 可显著减少不必要的重绘

@Observable
class UserViewModel {
    var name = ""      // 只有读取 name 的视图在 name 变化时更新
    var email = ""     // 读取 email 的视图不受 name 变化影响
    var isLoading = false
    
    func load() async {
        isLoading = true
        defer { isLoading = false }
        // Swift 6.2：默认在 MainActor，UI 更新安全
        let user = try? await api.fetchUser()
        name = user?.name ?? ""
        email = user?.email ?? ""
    }
}
```

func validateUsername() {
    let result = validate(username: "")
    #expect(result == .invalid("用户名不能为空"))
}

@Test("年龄验证", arguments: [0, 17, 18, 100, 150])
func validateAge(age: Int) {
    let result = validate(age: age)
    if age >= 18 && age <= 120 {
        #expect(result == .valid)
    } else {
        #expect(result == .invalid("年龄无效"))
    }
}
