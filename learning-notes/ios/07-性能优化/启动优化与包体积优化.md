# 启动优化与包体积优化

## 1. 冷启动 vs 热启动

```swift
// 冷启动阶段：
// 1. pre-main: dylib 加载 → rebase/binding → ObjC setup → initializer
// 2. main() → didFinishLaunching → 首屏渲染

// 测量启动时间：Xcode → Edit Scheme → Environment Variables
// DYLD_PRINT_STATISTICS = 1
// DYLD_PRINT_STATISTICS_DETAILS = 1
```

## 2. Pre-main 优化

```swift
// 减少动态库数量（合并或转为静态库）
// Podfile 中使用静态库
// use_frameworks! :linkage => :static

// 减少 +load 方法
// ❌ 在 +load 中执行初始化
// ✅ 使用 +initialize 或延迟初始化

// 减少 __attribute__((constructor))
// 将初始化逻辑移到首次使用时
class HeavyService {
    static let shared = HeavyService()  // 懒加载，首次访问时才初始化
    private init() { /* 耗时初始化 */ }
}
```

## 3. didFinishLaunching 优化

```swift
func application(_ application: UIApplication,
                 didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {
    // ✅ 只做必要初始化
    setupWindow()
    setupCrashReporting()

    // ✅ 延迟非必要初始化
    DispatchQueue.main.async {
        self.setupAnalytics()
        self.setupPushNotification()
    }

    // ✅ 后台初始化
    DispatchQueue.global().async {
        self.preloadDatabase()
        self.warmUpCache()
    }

    return true
}
```

## 4. 首屏渲染优化

```swift
// ✅ 骨架屏 / 占位图
class HomeViewController: UIViewController {
    override func viewDidLoad() {
        super.viewDidLoad()
        showSkeleton()  // 先显示骨架屏
        Task {
            let data = try await fetchHomeData()
            hideSkeleton()
            renderContent(data)
        }
    }
}

// ✅ 缓存首屏数据
func loadHomeData() async -> HomeData {
    // 先读缓存快速展示
    if let cached = CacheManager.shared.get("home_data", as: HomeData.self) {
        return cached
    }
    // 再请求网络更新
    let fresh = try? await api.fetchHomeData()
    if let fresh = fresh { CacheManager.shared.set("home_data", value: fresh) }
    return fresh ?? HomeData.empty
}
```

## 5. App Thinning

```swift
// 1. App Slicing: 按设备架构分发
// Build Settings → Build Active Architecture Only → Release: No
// Xcode 自动处理 @2x/@3x 资源分发

// 2. On-Demand Resources (ODR)
// 按需下载资源，减少初始包大小
let request = NSBundleResourceRequest(tags: ["level-5-assets"])
request.beginAccessingResources { error in
    if error == nil {
        // 资源可用
        let image = UIImage(named: "level5_bg")
    }
}
// 使用完毕释放
request.endAccessingResources()

// 3. Bitcode: 允许 Apple 重新优化二进制（已在 Xcode 14 移除）
```
