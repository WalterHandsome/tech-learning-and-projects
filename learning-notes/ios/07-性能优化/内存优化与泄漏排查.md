## 1. 循环引用检测

```swift
// ❌ Delegate 循环引用
class ParentVC: UIViewController {
    let child = ChildVC()
    override func viewDidLoad() {
        child.delegate = self  // child 强引用 parent
    }
}

// ✅ weak delegate
protocol ChildDelegate: AnyObject { func childDidFinish() }
class ChildVC: UIViewController {
    weak var delegate: ChildDelegate?  // weak 打破循环
}

// ❌ 闭包捕获 self
class ViewModel {
    var onUpdate: (() -> Void)?
    func setup() {
        onUpdate = { self.refresh() }  // 循环引用
    }
}

// ✅ [weak self] 或 [unowned self]
onUpdate = { [weak self] in self?.refresh() }
```

```swift
// Xcode → Debug → Debug Memory Graph
// 查看对象引用关系图，找到循环引用

// 常见泄漏场景
class NetworkManager {
    // ❌ NotificationCenter 未移除
    func setup() {
        NotificationCenter.default.addObserver(self, selector: #selector(handle),
                                                name: .someNotification, object: nil)
    }
    // ✅ 使用闭包 API（自动管理）
    private var observer: Any?
    func setup() {
        observer = NotificationCenter.default.addObserver(forName: .someNotification,
                                                           object: nil, queue: .main) { [weak self] _ in
            self?.handle()
        }
    }
    deinit { observer.map { NotificationCenter.default.removeObserver($0) } }
}
```

## 3. Combine 订阅泄漏

```swift
class ViewModel: ObservableObject {
    private var cancellables = Set<AnyCancellable>()

    // ✅ store(in:) 自动管理生命周期
    func bind() {
        publisher
            .sink { [weak self] value in self?.handle(value) }
            .store(in: &cancellables)
    }

    // ❌ 忘记存储 cancellable → 订阅立即取消
    func badBind() {
        publisher.sink { value in }  // 没有 store → 立即释放
    }
}
```

## 4. 图片内存优化

```swift
// ❌ 直接加载大图
let image = UIImage(named: "huge_photo")  // 可能占用几十MB

// ✅ 降采样（Downsampling）
func downsample(imageAt url: URL, to pointSize: CGSize, scale: CGFloat) -> UIImage? {
    let maxDimension = max(pointSize.width, pointSize.height) * scale
    let options: [CFString: Any] = [
        kCGImageSourceCreateThumbnailFromImageAlways: true,
        kCGImageSourceShouldCacheImmediately: true,
        kCGImageSourceCreateThumbnailWithTransform: true,
        kCGImageSourceThumbnailMaxPixelSize: maxDimension
    ]
    guard let source = CGImageSourceCreateWithURL(url as CFURL, nil),
          let cgImage = CGImageSourceCreateThumbnailAtIndex(source, 0, options as CFDictionary) else {
        return nil
    }
    return UIImage(cgImage: cgImage)
}
```

## 5. Autorelease Pool

```swift
// 循环中创建大量临时对象
func processLargeDataset() {
    for i in 0..<100000 {
        autoreleasepool {
            let data = generateData(index: i)
            processData(data)
            // data 在每次循环结束时释放，避免内存峰值
        }
    }
}
```

## 6. 内存警告处理

```swift
class ImageCache {
    static let shared = ImageCache()
    private var cache = NSCache<NSString, UIImage>()

    init() {
        cache.totalCostLimit = 100 * 1024 * 1024  // 100MB 上限
        // 监听内存警告
        NotificationCenter.default.addObserver(forName: UIApplication.didReceiveMemoryWarningNotification,
                                                object: nil, queue: .main) { [weak self] _ in
            self?.cache.removeAllObjects()
        }
    }
}
```
