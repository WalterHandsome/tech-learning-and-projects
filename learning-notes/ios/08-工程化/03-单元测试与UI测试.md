## 1. XCTest 基础

```swift
import XCTest
@testable import MyApp

class UserServiceTests: XCTestCase {

    var sut: UserService!  // System Under Test

    override func setUp() {
        super.setUp()
        sut = UserService(repository: MockUserRepository())
    }

    override func tearDown() {
        sut = nil
        super.tearDown()
    }

    func testFetchUsersReturnsNonEmpty() async throws {
        let users = try await sut.fetchUsers()
        XCTAssertFalse(users.isEmpty)
        XCTAssertEqual(users.count, 3)
    }

    func testFetchUserByIdReturnsCorrectUser() async throws {
        let user = try await sut.fetchUser(id: 1)
        XCTAssertEqual(user.name, "张三")
        XCTAssertEqual(user.email, "zhang@example.com")
    }

    func testFetchUserWithInvalidIdThrows() async {
        do {
            _ = try await sut.fetchUser(id: -1)
            XCTFail("应该抛出错误")
        } catch {
            XCTAssertTrue(error is UserError)
        }
    }
}
```

## 2. Mock 对象

```swift
protocol UserRepository {
    func getUsers() async throws -> [User]
    func getUser(id: Int) async throws -> User
}

class MockUserRepository: UserRepository {
    var mockUsers: [User] = [
        User(id: 1, name: "张三", email: "zhang@example.com"),
        User(id: 2, name: "李四", email: "li@example.com"),
        User(id: 3, name: "王五", email: "wang@example.com"),
    ]
    var shouldThrowError = false
    var fetchUsersCallCount = 0

    func getUsers() async throws -> [User] {
        fetchUsersCallCount += 1
        if shouldThrowError { throw UserError.networkError }
        return mockUsers
    }

    func getUser(id: Int) async throws -> User {
        guard let user = mockUsers.first(where: { $0.id == id }) else {
            throw UserError.notFound
        }
        return user
    }
}
```

## 3. ViewModel 测试

```swift
class UserListViewModelTests: XCTestCase {
    var viewModel: UserListViewModel!
    var mockRepo: MockUserRepository!

    override func setUp() {
        mockRepo = MockUserRepository()
        viewModel = UserListViewModel(repository: mockRepo)
    }

    func testLoadUsersSuccess() async {
        await viewModel.loadUsers()
        XCTAssertEqual(viewModel.users.count, 3)
        XCTAssertFalse(viewModel.isLoading)
        XCTAssertNil(viewModel.errorMessage)
    }

    func testLoadUsersFailure() async {
        mockRepo.shouldThrowError = true
        await viewModel.loadUsers()
        XCTAssertTrue(viewModel.users.isEmpty)
        XCTAssertNotNil(viewModel.errorMessage)
    }

    func testLoadUsersCallsRepository() async {
        await viewModel.loadUsers()
        XCTAssertEqual(mockRepo.fetchUsersCallCount, 1)
    }
}
```

## 5. XCUITest UI 测试

class LoginUITests: XCTestCase {
    let app = XCUIApplication()

        continueAfterFailure = false

## 6. Swift Testing 框架（Xcode 26+）

<!-- version-check: Swift Testing (Swift 6.2+, Xcode 26.4), checked 2026-04-29 -->

> 🔄 更新于 2026-04-29

Swift Testing 是 Apple 在 WWDC 2024 引入的现代测试框架，使用 `@Test` 宏替代 XCTest 的命名约定，默认并行执行测试。Xcode 26 / Swift 6.2 新增了 Exit Tests、测试附件和运行时问题检测。来源：[Swift Testing](https://developer.apple.com/xcode/swift-testing/)、[What's New in Testing](https://forums.swift.org/t/whats-new-in-testing-swift-6-2-xcode-26/80688)

### 6.1 基础用法

```swift
import Testing

// 使用 @Test 宏标记测试函数
@Test("用户创建成功")
func createUser() async throws {
    let service = UserService(repository: MockUserRepository())
    let user = try await service.createUser(name: "张三")
    #expect(user.name == "张三")
    #expect(user.id != nil)
}

// 使用 #expect 替代 XCTAssert 系列
@Test func fetchUsers() async throws {
    let users = try await service.fetchUsers()
    #expect(!users.isEmpty)
    #expect(users.count == 3)
}

// 使用 #require 替代 XCTUnwrap（失败时终止测试）
@Test func fetchUserById() async throws {
    let user = try #require(await service.fetchUser(id: 1))
    #expect(user.name == "张三")
}
```

### 6.2 参数化测试

```swift
// 参数化测试：一个测试函数覆盖多组输入
@Test("验证邮箱格式", arguments: [
    "test@example.com",
    "user.name@domain.org",
    "admin+tag@company.co"
])
func validEmail(_ email: String) {
    #expect(EmailValidator.isValid(email))
}

// 多参数组合
@Test(arguments: ["admin", "user"], [true, false])
func checkPermission(role: String, isActive: Bool) {
    let result = PermissionChecker.canAccess(role: role, active: isActive)
    if role == "admin" && isActive {
        #expect(result == true)
    }
}
```

### 6.3 Traits（测试特征）

```swift
// 标记测试分类
@Test(.tags(.networking))
func apiCallTest() async throws { ... }

// 设置超时
@Test(.timeLimit(.minutes(2)))
func longRunningTest() async throws { ... }

// 条件跳过
@Test(.enabled(if: ProcessInfo.processInfo.environment["CI"] != nil))
func ciOnlyTest() { ... }

// 标记为已知问题（不会导致测试套件失败）
@Test(.bug("https://github.com/org/repo/issues/123"))
func knownIssueTest() { ... }

// 自定义 Tag
extension Tag {
    @Tag static var networking: Self
    @Tag static var database: Self
    @Tag static var ui: Self
}
```

### 6.4 Exit Tests（Swift 6.2 新增）

```swift
// 测试 precondition / fatalError 是否正确触发
@Test func divisionByZeroCrashes() async {
    await #expect(exitsWith: .failure) {
        // 这段代码应该触发 fatalError
        _ = divide(10, by: 0)
    }
}

// 测试 preconditionFailure
@Test func invalidIndexCrashes() async {
    await #expect(exitsWith: .failure) {
        let array = [1, 2, 3]
        _ = array[10]  // 越界访问
    }
}
```

### 6.5 Attachments（Swift 6.2 新增）

> 🔄 更新于 2026-04-29

测试附件允许在测试失败时自动捕获诊断数据（截图、日志、数据快照），附件会出现在 Xcode 测试报告中。来源：[Swift Testing Attachments Proposal](https://github.com/swiftlang/swift-evolution/blob/main/proposals/testing/0010-attachments.md)

```swift
import Testing

@Test func validateUserProfile() async throws {
    let profile = try await fetchProfile(id: 42)
    
    // 附加 JSON 数据用于调试
    let jsonData = try JSONEncoder().encode(profile)
    Attachment(data: jsonData, named: "profile.json").attach()
    
    #expect(profile.name == "张三")
}

// 附加截图（UI 测试场景）
@Test func screenshotOnFailure() async throws {
    let screenshot = takeScreenshot()
    Attachment(data: screenshot.pngData()!, named: "failure.png").attach()
    
    #expect(isLoginScreenVisible)
}
```

### 6.6 Runtime Issue Detection（Xcode 26 新增）

> 🔄 更新于 2026-04-29

Xcode 26 引入运行时问题检测，在测试执行期间自动捕获线程安全违规、数据竞争等运行时问题，默认以警告形式报告。来源：[What's New in Testing](https://rachelbrindle.com/2025/06/26/whats-new-in-testing-swift-6-2/)

```swift
// 运行时问题默认报告为警告
// 可在 XCTest 中自定义严重级别
class StrictRuntimeTests: XCTestCase {
    override func recordIssue(_ issue: XCTIssue) {
        var modifiedIssue = issue
        // 将运行时问题从警告升级为错误
        if issue.compactDescription.contains("runtime") {
            modifiedIssue.severity = .error
        }
        super.recordIssue(modifiedIssue)
    }
}
```

### 6.7 Testing Workgroup

Swift Testing Workgroup 于 2025-03 正式成立，负责治理 Swift Testing 和 Corelibs XCTest 项目。当前活跃提案包括 Issue Handling Traits（自定义问题处理）。

### 6.8 Swift Testing vs XCTest 对比

```
特性              Swift Testing          XCTest
──────────────────────────────────────────────────
测试标记          @Test 宏               test 前缀命名
断言             #expect / #require      XCTAssert 系列
参数化测试        原生支持               不支持
并行执行          默认并行               需配置
测试组织          @Suite + .tags         XCTestCase 类
Exit Tests       ✅ 支持                ❌ 不支持
UI 测试          ❌ 用 XCTest           ✅ XCUITest
性能测试          ❌ 用 XCTest           ✅ measure {}
推荐场景          单元测试 + 逻辑测试     UI 测试 + 性能测试
```

> **迁移建议**：新项目的单元测试和逻辑测试使用 Swift Testing，UI 测试和性能测试继续使用 XCTest。两者可以在同一项目中共存。
