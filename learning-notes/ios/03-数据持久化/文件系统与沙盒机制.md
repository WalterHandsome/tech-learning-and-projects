## 2. FileManager 文件操作

```swift
let fm = FileManager.default
let docsURL = fm.urls(for: .documentDirectory, in: .userDomainMask).first!

// 创建目录
let folderURL = docsURL.appendingPathComponent("Images")
try fm.createDirectory(at: folderURL, withIntermediateDirectories: true)

// 写入文件
let fileURL = docsURL.appendingPathComponent("note.txt")
let text = "Hello, 文件系统"
try text.write(to: fileURL, atomically: true, encoding: .utf8)

// 读取文件
let content = try String(contentsOf: fileURL, encoding: .utf8)

// 检查文件是否存在
if fm.fileExists(atPath: fileURL.path) {
    print("文件存在")
}

// 复制 / 移动 / 删除
try fm.copyItem(at: sourceURL, to: destURL)
try fm.moveItem(at: sourceURL, to: destURL)
try fm.removeItem(at: fileURL)

// 列出目录内容
let files = try fm.contentsOfDirectory(at: docsURL, includingPropertiesForKeys: nil)
```

## 3. JSON 文件读写

```swift
struct AppConfig: Codable {
    var apiBaseURL: String
    var maxRetries: Int
    var features: [String: Bool]
}

class ConfigManager {
    private let fileURL: URL

    init() {
        let docs = FileManager.default.urls(for: .documentDirectory, in: .userDomainMask).first!
        fileURL = docs.appendingPathComponent("config.json")
    }

    func save(_ config: AppConfig) throws {
        let data = try JSONEncoder().encode(config)
        try data.write(to: fileURL)
    }

    func load() throws -> AppConfig {
        let data = try Data(contentsOf: fileURL)
        return try JSONDecoder().decode(AppConfig.self, from: data)
    }
}
```

    let url = FileManager.default.urls(for: .documentDirectory, in: .userDomainMask).first!

## 5. 文件大小与磁盘空间

```swift
// 获取文件大小
func fileSize(at url: URL) -> Int64? {
    let attributes = try? FileManager.default.attributesOfItem(atPath: url.path)
    return attributes?[.size] as? Int64
}

// 计算目录大小
func directorySize(at url: URL) -> Int64 {
    let fm = FileManager.default
    guard let enumerator = fm.enumerator(at: url, includingPropertiesForKeys: [.fileSizeKey]) else { return 0 }
    var total: Int64 = 0
    for case let fileURL as URL in enumerator {
        let size = (try? fileURL.resourceValues(forKeys: [.fileSizeKey]))?.fileSize ?? 0
        total += Int64(size)
    }
    return total
}
```
