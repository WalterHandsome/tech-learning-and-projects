## 1. 构造器注入（推荐）

```swift
// 协议定义
protocol UserRepository {
    func getUsers() async throws -> [User]
}

protocol AuthService {
    func login(email: String, password: String) async throws -> Token
}

// 实现
class RemoteUserRepository: UserRepository {
    private let apiClient: APIClient
    init(apiClient: APIClient) { self.apiClient = apiClient }

    func getUsers() async throws -> [User] {
        try await apiClient.request("/users")
    }
}

// ViewModel 通过构造器注入依赖
class UserListViewModel: ObservableObject {
    @Published var users: [User] = []
    private let repository: UserRepository

    init(repository: UserRepository) {
        self.repository = repository
    }

    func load() async {
        users = (try? await repository.getUsers()) ?? []
    }
}

// 使用
let vm = UserListViewModel(repository: RemoteUserRepository(apiClient: .shared))
```

## 3. DI 容器（Swinject）

```swift
import Swinject

let container = Container()

// 注册依赖
container.register(APIClient.self) { _ in APIClient.shared }
container.register(UserRepository.self) { r in
    RemoteUserRepository(apiClient: r.resolve(APIClient.self)!)
}
container.register(UserListViewModel.self) { r in
    UserListViewModel(repository: r.resolve(UserRepository.self)!)
}

// 解析
let viewModel = container.resolve(UserListViewModel.self)!

// Assembly 模块化注册
class NetworkAssembly: Assembly {
    func assemble(container: Container) {
        container.register(APIClient.self) { _ in APIClient.shared }.inObjectScope(.container)
        container.register(UserRepository.self) { r in
            RemoteUserRepository(apiClient: r.resolve(APIClient.self)!)
        }
    }
}

let assembler = Assembler([NetworkAssembly(), ViewModelAssembly()])
```

## 4. 测试中的 Mock 注入

```swift
class MockUserRepository: UserRepository {
    var mockUsers: [User] = []
    var shouldThrow = false

    func getUsers() async throws -> [User] {
        if shouldThrow { throw NSError(domain: "test", code: -1) }
        return mockUsers
    }
}

// 单元测试
func testFetchUsers() async {
    let mock = MockUserRepository()
    mock.mockUsers = [User(id: 1, name: "测试用户", email: "test@test.com")]
    let vm = UserListViewModel(repository: mock)
    await vm.load()
    XCTAssertEqual(vm.users.count, 1)
}
```

## 5. SPM 模块化

```swift
// Package.swift
// swift-tools-version: 5.9
import PackageDescription

let package = Package(
    name: "MyAppModules",
    platforms: [.iOS(.v16)],
    products: [
        .library(name: "Domain", targets: ["Domain"]),
        .library(name: "Data", targets: ["Data"]),
        .library(name: "Presentation", targets: ["Presentation"]),
    ],
    targets: [
        .target(name: "Domain"),                              // 纯业务逻辑，无外部依赖
        .target(name: "Data", dependencies: ["Domain"]),      // 数据层依赖 Domain
        .target(name: "Presentation", dependencies: ["Domain"]), // UI 层依赖 Domain
        .testTarget(name: "DomainTests", dependencies: ["Domain"]),
    ]
)
```

## 6. Environment 注入（SwiftUI）

```swift
// 定义 EnvironmentKey
struct UserRepositoryKey: EnvironmentKey {
    static let defaultValue: UserRepository = RemoteUserRepository(apiClient: .shared)
}

extension EnvironmentValues {
    var userRepository: UserRepository {
        get { self[UserRepositoryKey.self] }
        set { self[UserRepositoryKey.self] = newValue }
    }
}

// 注入
ContentView()
    .environment(\.userRepository, MockUserRepository())

// 使用
struct UserView: View {
    @Environment(\.userRepository) var repository
}
```
