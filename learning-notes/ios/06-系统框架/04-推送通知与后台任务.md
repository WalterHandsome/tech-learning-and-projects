## 1. 请求通知权限

```swift
import UserNotifications

func requestNotificationPermission() {
    UNUserNotificationCenter.current().requestAuthorization(options: [.alert, .sound, .badge]) { granted, error in
        if granted { print("通知权限已授权") }
        DispatchQueue.main.async {
            UIApplication.shared.registerForRemoteNotifications()
        }
    }
}
```

## 2. 本地通知

```swift
func scheduleLocalNotification() {
    let content = UNMutableNotificationContent()
    content.title = "提醒"
    content.body = "该喝水了"
    content.sound = .default
    content.badge = 1

    // 时间触发
    let trigger = UNTimeIntervalNotificationTrigger(timeInterval: 60, repeats: false)

    // 日期触发
    var dateComponents = DateComponents()
    dateComponents.hour = 9
    dateComponents.minute = 0
    let dateTrigger = UNCalendarNotificationTrigger(dateMatching: dateComponents, repeats: true)

    let request = UNNotificationRequest(identifier: "water-reminder", content: content, trigger: trigger)
    UNUserNotificationCenter.current().add(request)
}

// 取消通知
UNUserNotificationCenter.current().removePendingNotificationRequests(withIdentifiers: ["water-reminder"])
```

## 4. 处理通知响应

```swift
class NotificationDelegate: NSObject, UNUserNotificationCenterDelegate {
    // 前台收到通知
    func userNotificationCenter(_ center: UNUserNotificationCenter,
                                willPresent notification: UNNotification,
                                withCompletionHandler completionHandler: @escaping (UNNotificationPresentationOptions) -> Void) {
        completionHandler([.banner, .sound, .badge])
    }

    // 用户点击通知
    func userNotificationCenter(_ center: UNUserNotificationCenter,
                                didReceive response: UNNotificationResponse,
                                withCompletionHandler completionHandler: @escaping () -> Void) {
        let userInfo = response.notification.request.content.userInfo
        handleNotification(userInfo)
        completionHandler()
    }
}
```

## 5. BGTaskScheduler 后台任务

```swift
import BackgroundTasks

// 注册后台任务（AppDelegate didFinishLaunching）
BGTaskScheduler.shared.register(forTaskWithIdentifier: "com.app.refresh", using: nil) { task in
    handleAppRefresh(task: task as! BGAppRefreshTask)
}

// 调度任务
func scheduleAppRefresh() {
    let request = BGAppRefreshTaskRequest(identifier: "com.app.refresh")
    request.earliestBeginDate = Date(timeIntervalSinceNow: 15 * 60)  // 最早15分钟后
    try? BGTaskScheduler.shared.submit(request)
}

// 处理后台任务
func handleAppRefresh(task: BGAppRefreshTask) {
    scheduleAppRefresh()  // 调度下一次

    let operation = RefreshOperation()
    task.expirationHandler = { operation.cancel() }

    operation.completionBlock = {
        task.setTaskCompleted(success: !operation.isCancelled)
    }
    OperationQueue().addOperation(operation)
}
```

### 7.1 推送通知变化

iOS 26 对推送通知引入了多项改进，重点在智能分类和优先级管理：

**AI 驱动的通知智能排序**：系统自动对通知进行优先级排序，重要通知（如时间敏感的交易消息）会优先展示，低优先级通知可能被折叠到摘要中。

**通知屏幕闪烁辅助功能**（iOS 26.2）：新增屏幕闪烁提示，增强无障碍访问体验。

**UIScene 生命周期迁移**：iOS 26 开始在控制台输出警告，要求应用采用 `UISceneDelegate` 生命周期。iOS 27 将强制要求。这影响推送通知的处理方式——通知回调需要在 Scene 级别处理。

来源：[OneSignal iOS 26 Recap](https://onesignal.com/blog/wwdc-2025-new-release-recap/)、[Courier UIScene 迁移指南](https://courier.com/blog/ios-26-push-notification-changes-uiscene-requirment-ios-27)

```swift
// iOS 26+ 推荐：在 SceneDelegate 中处理通知
class SceneDelegate: UIResponder, UIWindowSceneDelegate {
    var window: UIWindow?

    func scene(_ scene: UIScene, willConnectTo session: UISceneSession,
               options connectionOptions: UIScene.ConnectionOptions) {
        // 处理冷启动时的通知
        if let response = connectionOptions.notificationResponse {
            handleNotification(response.notification.request.content.userInfo)
        }
    }
}

// SwiftUI 中处理通知
@main
struct MyApp: App {
    @UIApplicationDelegateAdaptor(AppDelegate.self) var appDelegate

    var body: some Scene {
        WindowGroup {
            ContentView()
                .onOpenURL { url in
                    // 处理通知中的深链接
                }
        }
    }
}
```

### 7.2 BGContinuedProcessingTask（iOS 26 新增）

iOS 26 引入 `BGContinuedProcessingTask`，允许应用在从前台切换到后台时继续完成用户发起的长时间任务（如视频导出、文件处理），无需用户保持应用在前台。

来源：[WWDC25 Session: Finish tasks in the background](https://developer.apple.com/videos/play/wwdc2025/227/)

```swift
import BackgroundTasks

// 注册后台继续处理任务
BGTaskScheduler.shared.register(
    forTaskWithIdentifier: "com.app.videoExport",
    using: nil
) { task in
    guard let continuedTask = task as? BGContinuedProcessingTask else { return }
    handleContinuedProcessing(task: continuedTask)
}

// 在前台开始任务，切换到后台时自动继续
func startVideoExport(videoURL: URL) {
    let request = BGContinuedProcessingTaskRequest(
        identifier: "com.app.videoExport"
    )
    // 系统会在应用进入后台时自动接管任务
    try? BGTaskScheduler.shared.submit(request)

    // 开始实际的导出工作
    exportVideo(from: videoURL) { progress in
        // 更新进度（前台和后台都会执行）
        updateProgress(progress)
    }
}

func handleContinuedProcessing(task: BGContinuedProcessingTask) {
    task.expirationHandler = {
        // 系统要求停止时的清理逻辑
        savePartialProgress()
    }

    // 任务完成后通知系统
    task.setTaskCompleted(success: true)
}
```

**BGContinuedProcessingTask vs 其他后台 API 对比**：

| API | 适用场景 | 运行时机 | iOS 版本 |
|-----|---------|---------|---------|
| `BGAppRefreshTask` | 定期数据刷新 | 系统调度（不可预测） | iOS 13+ |
| `BGProcessingTask` | 大型数据处理 | 设备充电/空闲时 | iOS 13+ |
| `BGContinuedProcessingTask` | 用户发起的长任务 | 前台→后台过渡时 | **iOS 26+** |
| `beginBackgroundTask` | 短时后台完成 | 进入后台后约 30 秒 | iOS 4+ |
| Background URLSession | 大文件下载/上传 | 系统管理 | iOS 7+ |

### 7.3 Background GPU Access（iOS 26 新增）

iOS 26 还引入了后台 GPU 访问能力，配合 `BGContinuedProcessingTask` 使用，允许视频处理、图像渲染等 GPU 密集型任务在后台继续执行。需要在 Xcode 中启用 "Background GPU Access" capability。
