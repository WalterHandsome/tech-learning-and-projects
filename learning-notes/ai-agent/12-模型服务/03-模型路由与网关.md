## 1. LiteLLM 统一接口

LiteLLM 提供统一的 API 接口，支持 100+ 模型提供商，一套代码切换不同模型。

```python
from litellm import completion

# 统一接口调用不同模型
# OpenAI
response = completion(
    model="gpt-4o",
    messages=[{"role": "user", "content": "你好"}],
)

# Claude
response = completion(
    model="claude-sonnet-4-6-20260217",
    messages=[{"role": "user", "content": "你好"}],
)

# 本地 Ollama
response = completion(
    model="ollama/qwen2.5:14b",
    messages=[{"role": "user", "content": "你好"}],
    api_base="http://localhost:11434",
)

# Azure OpenAI
response = completion(
    model="azure/gpt-4o",
    messages=[{"role": "user", "content": "你好"}],
    api_base="https://my-endpoint.openai.azure.com",
    api_key="azure-key",
)
```

## 2. LiteLLM Proxy（模型网关）

```yaml
# litellm_config.yaml
model_list:
  - model_name: gpt-4o
    litellm_params:
      model: openai/gpt-4o
      api_key: sk-xxx

  - model_name: gpt-4o
    litellm_params:
      model: azure/gpt-4o-deployment
      api_base: https://my-azure.openai.azure.com
      api_key: azure-xxx

  - model_name: claude-sonnet
    litellm_params:
      model: claude-sonnet-4-6-20260217
      api_key: sk-ant-xxx

  - model_name: local-qwen
    litellm_params:
      model: ollama/qwen2.5:14b
      api_base: http://localhost:11434

router_settings:
  routing_strategy: least-busy  # simple-shuffle / least-busy / latency-based
  num_retries: 3
  timeout: 60
```

```bash
# 启动 Proxy
litellm --config litellm_config.yaml --port 4000

# 所有应用统一访问 Proxy
curl http://localhost:4000/v1/chat/completions \
  -H "Authorization: Bearer sk-proxy-key" \
  -d '{"model": "gpt-4o", "messages": [{"role": "user", "content": "hello"}]}'
```

## 3. 模型负载均衡

```python
from litellm import Router

# 配置多个模型实例
router = Router(
    model_list=[
        {
            "model_name": "gpt-4o",
            "litellm_params": {"model": "openai/gpt-4o", "api_key": "key-1"},
            "tpm": 100000,  # 每分钟 token 限制
            "rpm": 500,     # 每分钟请求限制
        },
        {
            "model_name": "gpt-4o",
            "litellm_params": {"model": "azure/gpt-4o", "api_key": "key-2"},
            "tpm": 200000,
            "rpm": 1000,
        },
    ],
    routing_strategy="least-busy",
    num_retries=3,
    fallbacks=[{"gpt-4o": ["claude-sonnet"]}],  # 失败时降级
)

# 自动路由到最空闲的实例
response = await router.acompletion(
    model="gpt-4o",
    messages=[{"role": "user", "content": "你好"}],
)
```

## 4. Fallback 策略

```python
from litellm import completion
import litellm

# 方式一：LiteLLM 内置 fallback
litellm.set_verbose = True

response = completion(
    model="gpt-4o",
    messages=[{"role": "user", "content": "你好"}],
    fallbacks=["claude-sonnet-4-6-20260217", "gpt-4o-mini"],  # 依次降级
    num_retries=2,
)

# 方式二：自定义 fallback 逻辑
class ModelFallback:
    def __init__(self, models: list[str]):
        self.models = models

    async def call(self, messages: list[dict]) -> str:
        for model in self.models:
            try:
                response = await litellm.acompletion(
                    model=model, messages=messages, timeout=30,
                )
                return response.choices[0].message.content
            except Exception as e:
                print(f"{model} 失败: {e}, 尝试下一个模型")
        raise Exception("所有模型均不可用")

fallback = ModelFallback(["gpt-4o", "claude-sonnet-4-6-20260217", "ollama/qwen2.5:14b"])
result = await fallback.call([{"role": "user", "content": "你好"}])
```

## 5. Token 用量追踪

```python
import litellm
from litellm import completion_cost

# 自动追踪成本
litellm.success_callback = ["langfuse"]  # 发送到 LangFuse

response = completion(model="gpt-4o", messages=[{"role": "user", "content": "你好"}])

# 计算单次成本
cost = completion_cost(completion_response=response)
print(f"本次调用成本: ${cost:.6f}")

# 自定义用量追踪
class UsageTracker:
    def __init__(self):
        self.records: list[dict] = []

    def track(self, model: str, usage: dict, cost: float):
        self.records.append({
            "model": model,
            "input_tokens": usage.get("prompt_tokens", 0),
            "output_tokens": usage.get("completion_tokens", 0),
            "cost": cost,
        })

    def summary(self) -> dict:
        total_cost = sum(r["cost"] for r in self.records)
        by_model = {}
        for r in self.records:
            m = r["model"]
            by_model.setdefault(m, {"calls": 0, "cost": 0})
            by_model[m]["calls"] += 1
            by_model[m]["cost"] += r["cost"]
        return {"total_cost": total_cost, "by_model": by_model}
```

## 6. API Key 管理

```python
import os
from dataclasses import dataclass

@dataclass
class APIKeyConfig:
    """API Key 管理配置"""
    provider: str
    key: str
    rate_limit: int  # RPM
    monthly_budget: float

class APIKeyManager:
    def __init__(self):
        self.keys: list[APIKeyConfig] = []
        self.usage: dict[str, float] = {}

    def add_key(self, config: APIKeyConfig):
        self.keys.append(config)

    def get_available_key(self, provider: str) -> str:
        """获取可用的 API Key（轮询 + 预算检查）"""
        candidates = [k for k in self.keys if k.provider == provider]
        for key_config in candidates:
            spent = self.usage.get(key_config.key, 0)
            if spent < key_config.monthly_budget:
                return key_config.key
        raise Exception(f"所有 {provider} API Key 已达预算上限")

manager = APIKeyManager()
manager.add_key(APIKeyConfig("openai", os.getenv("OPENAI_KEY_1"), 500, 100.0))
manager.add_key(APIKeyConfig("openai", os.getenv("OPENAI_KEY_2"), 500, 100.0))
```
