## 1. 工具链编排

```python
from typing import Callable

class ToolChain:
    """工具链编排器：按顺序执行多个工具"""

    def __init__(self):
        self.steps: list[dict] = []

    def add_step(self, tool: Callable, name: str, depends_on: list[str] = None):
        self.steps.append({"tool": tool, "name": name, "depends_on": depends_on or []})

    async def execute(self, initial_input: dict) -> dict:
        results = {}
        for step in self.steps:
            # 收集依赖结果作为输入
            step_input = {**initial_input}
            for dep in step["depends_on"]:
                step_input[dep] = results[dep]

            results[step["name"]] = await step["tool"](**step_input)
        return results

# 使用示例
chain = ToolChain()
chain.add_step(search_docs, "search")
chain.add_step(rerank_results, "rerank", depends_on=["search"])
chain.add_step(generate_answer, "answer", depends_on=["rerank"])
result = await chain.execute({"query": "MCP 协议是什么？"})
```

## 2. 权限控制

```python
from enum import Enum
from functools import wraps

class Permission(Enum):
    READ = "read"
    WRITE = "write"
    EXECUTE = "execute"
    ADMIN = "admin"

class ToolPermissionManager:
    def __init__(self):
        self.tool_permissions: dict[str, set[Permission]] = {}
        self.user_permissions: dict[str, set[Permission]] = {}

    def register_tool(self, tool_name: str, required: set[Permission]):
        self.tool_permissions[tool_name] = required

    def check_permission(self, user_id: str, tool_name: str) -> bool:
        required = self.tool_permissions.get(tool_name, set())
        user_perms = self.user_permissions.get(user_id, set())
        return required.issubset(user_perms)

pm = ToolPermissionManager()
pm.register_tool("query_database", {Permission.READ})
pm.register_tool("delete_record", {Permission.WRITE, Permission.ADMIN})

def require_permission(*permissions: Permission):
    """权限装饰器"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, user_id: str, **kwargs):
            user_perms = pm.user_permissions.get(user_id, set())
            if not set(permissions).issubset(user_perms):
                raise PermissionError(f"用户 {user_id} 缺少权限: {permissions}")
            return await func(*args, **kwargs)
        return wrapper
    return decorator

@require_permission(Permission.READ)
async def query_data(sql: str) -> list:
    return db.execute(sql).fetchall()
```

## 3. 输入验证

```python
from pydantic import BaseModel, Field, field_validator

class SQLQueryInput(BaseModel):
    """SQL 查询输入验证"""
    sql: str = Field(..., max_length=1000)
    database: str = Field(default="main")
    timeout: int = Field(default=30, ge=1, le=300)

    @field_validator("sql")
    @classmethod
    def validate_sql(cls, v: str) -> str:
        forbidden = ["DROP", "DELETE", "TRUNCATE", "ALTER", "GRANT"]
        upper = v.upper()
        for kw in forbidden:
            if kw in upper:
                raise ValueError(f"禁止使用 {kw} 语句")
        if not upper.strip().startswith("SELECT"):
            raise ValueError("只允许 SELECT 查询")
        return v

class FileOperationInput(BaseModel):
    """文件操作输入验证"""
    path: str
    operation: str = Field(..., pattern="^(read|list)$")

    @field_validator("path")
    @classmethod
    def validate_path(cls, v: str) -> str:
        if ".." in v or v.startswith("/"):
            raise ValueError("不允许路径遍历")
        return v
```

## 4. 速率限制

```python
import time
from collections import defaultdict

class RateLimiter:
    """滑动窗口速率限制器"""

    def __init__(self, max_calls: int, window_seconds: int):
        self.max_calls = max_calls
        self.window = window_seconds
        self.calls: dict[str, list[float]] = defaultdict(list)

    def check(self, key: str) -> bool:
        now = time.time()
        # 清理过期记录
        self.calls[key] = [t for t in self.calls[key] if now - t < self.window]
        if len(self.calls[key]) >= self.max_calls:
            return False
        self.calls[key].append(now)
        return True

# 每用户每分钟最多 20 次工具调用
limiter = RateLimiter(max_calls=20, window_seconds=60)

async def execute_tool(user_id: str, tool_name: str, args: dict):
    if not limiter.check(f"{user_id}:{tool_name}"):
        raise Exception("调用频率超限，请稍后重试")
    return await tools[tool_name](**args)
```

## 5. 审计日志

```python
import logging
from datetime import datetime

audit_logger = logging.getLogger("tool_audit")
audit_logger.setLevel(logging.INFO)
handler = logging.FileHandler("tool_audit.log")
handler.setFormatter(logging.Formatter("%(asctime)s - %(message)s"))
audit_logger.addHandler(handler)

class ToolAuditor:
    """工具调用审计"""

    @staticmethod
    def log_call(user_id: str, tool_name: str, args: dict, result: any, duration_ms: float):
        audit_logger.info(
            f"user={user_id} tool={tool_name} args={args} "
            f"status=success duration={duration_ms:.1f}ms"
        )

    @staticmethod
    def log_error(user_id: str, tool_name: str, args: dict, error: str):
        audit_logger.error(
            f"user={user_id} tool={tool_name} args={args} "
            f"status=error error={error}"
        )
```

## 6. 沙箱执行

```python
import subprocess
import tempfile
import os

class CodeSandbox:
    """安全的代码执行沙箱"""

    def __init__(self, timeout: int = 30, max_memory_mb: int = 256):
        self.timeout = timeout
        self.max_memory = max_memory_mb

    def execute_python(self, code: str) -> dict:
        """在隔离环境中执行 Python 代码"""
        with tempfile.NamedTemporaryFile(mode="w", suffix=".py", delete=False) as f:
            f.write(code)
            f.flush()

            try:
                result = subprocess.run(
                    ["python", f.name],
                    capture_output=True, text=True,
                    timeout=self.timeout,
                    env={**os.environ, "PYTHONDONTWRITEBYTECODE": "1"},
                    cwd=tempfile.gettempdir(),
                )
                return {
                    "stdout": result.stdout[:5000],
                    "stderr": result.stderr[:2000],
                    "returncode": result.returncode,
                }
            except subprocess.TimeoutExpired:
                return {"error": f"执行超时（{self.timeout}s）"}
            finally:
                os.unlink(f.name)

sandbox = CodeSandbox(timeout=10)
result = sandbox.execute_python("print(sum(range(100)))")
```
## 7. 效用引导的 Agent 编排决策

> 🔄 更新于 2026-04-21

<!-- version-check: arXiv 2603.19896 Utility-Guided Orchestration, checked 2026-04-21 -->

arXiv 论文 [Utility-Guided Agent Orchestration](https://arxiv.org/html/2603.19896v1) 提出了一种基于效用理论的 Agent 编排策略，解决"Agent 什么时候该调用工具、什么时候该直接回答"这个核心问题。

**核心思想**：在每个决策点，Agent 从 5 种动作中选择，通过平衡四个维度做出最优决策：

```
Agent 编排决策空间：

  动作选项                    决策维度
  ┌──────────┐              ┌──────────────┐
  │ respond  │ 直接回答      │ 预估收益 Gain │
  │ retrieve │ 检索上下文    │ 步骤成本 Cost │
  │ tool_call│ 调用工具      │ 不确定性 Unc  │
  │ verify   │ 验证结果      │ 冗余度 Red    │
  │ stop     │ 终止执行      └──────────────┘
  └──────────┘

  效用函数：U(action) = Gain - Cost - λ₁·Uncertainty + λ₂·(1 - Redundancy)
```

**实际应用**：这个框架对降低 Agent 的 API 调用成本和延迟有直接指导意义。例如，当 Agent 已经有足够信息回答问题时，效用函数会倾向于 `respond` 而非继续 `tool_call`，避免不必要的工具调用。

与近期其他工具使用优化方法的对比：

| 方法 | 路径 | 核心机制 | 来源 |
|------|------|---------|------|
| Utility-Guided | 效用理论 | 多维度效用函数决策 | arXiv 2603.19896 |
| EvoTool | 进化算法 | 归因变异 + 多样性选择 | arXiv 2603.04900 |
| RLTR | 强化学习 | 工具使用完整性奖励 | arXiv 2508.19598 |
