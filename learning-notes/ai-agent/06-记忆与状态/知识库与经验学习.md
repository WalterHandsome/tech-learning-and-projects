## 1. 知识库构建

```python
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings
from langchain.text_splitter import RecursiveCharacterTextSplitter
from pathlib import Path
import hashlib

class KnowledgeBase:
    """Agent 知识库管理"""

    def __init__(self, persist_dir: str = "./knowledge_db"):
        self.embeddings = OpenAIEmbeddings(model="text-embedding-3-small")
        self.store = Chroma(
            collection_name="knowledge",
            embedding_function=self.embeddings,
            persist_directory=persist_dir,
        )
        self.splitter = RecursiveCharacterTextSplitter(chunk_size=500, chunk_overlap=50)

    def add_document(self, content: str, source: str, metadata: dict = None):
        """添加文档到知识库"""
        doc_hash = hashlib.md5(content.encode()).hexdigest()
        chunks = self.splitter.split_text(content)
        metadatas = [
            {**(metadata or {}), "source": source, "doc_hash": doc_hash, "chunk_idx": i}
            for i in range(len(chunks))
        ]
        self.store.add_texts(texts=chunks, metadatas=metadatas)

    def update_document(self, source: str, new_content: str):
        """更新已有文档（删除旧版本，添加新版本）"""
        # 删除旧文档
        self.store._collection.delete(where={"source": source})
        # 添加新文档
        self.add_document(new_content, source)

    def search(self, query: str, k: int = 5, filter_dict: dict = None) -> list:
        kwargs = {"k": k}
        if filter_dict:
            kwargs["filter"] = filter_dict
        return self.store.similarity_search(query, **kwargs)

kb = KnowledgeBase()
kb.add_document("MCP 是 Anthropic 提出的协议...", source="mcp-guide.md")
results = kb.search("MCP 协议是什么？")
```

## 2. 经验回放（Experience Replay）

```python
from datetime import datetime
import json

class ExperienceStore:
    """Agent 经验存储：记录成功和失败的执行经验"""

    def __init__(self, vector_store):
        self.store = vector_store

    def record_experience(self, task: str, plan: str, result: str, success: bool, feedback: str = ""):
        """记录一次执行经验"""
        experience = {
            "task": task,
            "plan": plan,
            "result": result,
            "success": success,
            "feedback": feedback,
            "timestamp": datetime.now().isoformat(),
        }
        # 存储为向量，便于语义检索
        text = f"任务：{task}\n方案：{plan}\n结果：{'成功' if success else '失败'}\n经验：{feedback}"
        self.store.add_texts(
            texts=[text],
            metadatas=[experience],
        )

    def recall_similar(self, task: str, k: int = 3, success_only: bool = True) -> list:
        """检索类似任务的历史经验"""
        filter_dict = {"success": True} if success_only else None
        return self.store.similarity_search(task, k=k, filter=filter_dict)

# 使用经验指导新任务
exp_store = ExperienceStore(vector_store)
exp_store.record_experience(
    task="分析销售数据",
    plan="1.加载CSV 2.清洗数据 3.统计分析 4.可视化",
    result="成功生成分析报告",
    success=True,
    feedback="使用 pandas groupby 比循环快 10 倍",
)
```

## 3. 反思机制（Reflection）

```python
class ReflectionAgent:
    """自反思 Agent：执行后反思并改进"""

    def __init__(self, llm):
        self.llm = llm
        self.reflections: list[str] = []

    async def execute_with_reflection(self, task: str, max_iterations: int = 3) -> str:
        result = ""
        for i in range(max_iterations):
            # 执行
            if i == 0:
                result = await self._execute(task)
            else:
                result = await self._execute(task, previous=result, reflection=reflection)

            # 反思
            reflection = await self._reflect(task, result)

            if "满意" in reflection or "无需改进" in reflection:
                break

            self.reflections.append(reflection)
        return result

    async def _execute(self, task: str, previous: str = "", reflection: str = "") -> str:
        prompt = f"任务：{task}"
        if previous:
            prompt += f"\n上次结果：{previous}\n改进建议：{reflection}\n请根据建议改进。"
        return (await self.llm.ainvoke(prompt)).content

    async def _reflect(self, task: str, result: str) -> str:
        prompt = f"""请反思以下任务执行结果：

任务：{task}
结果：{result}

评估维度：
1. 完整性：是否完整回答了问题？
2. 准确性：信息是否准确？
3. 可改进点：有哪些可以改进的地方？

如果结果已经很好，请回复"满意，无需改进"。"""
        return (await self.llm.ainvoke(prompt)).content
```

## 4. 自我改进

```python
class SelfImprovingAgent:
    """自我改进 Agent：从反馈中学习"""

    def __init__(self, llm, experience_store: ExperienceStore):
        self.llm = llm
        self.exp_store = experience_store
        self.learned_rules: list[str] = []

    async def learn_from_feedback(self, task: str, result: str, feedback: str):
        """从用户反馈中提取规则"""
        prompt = f"""基于以下反馈，提取一条通用规则供未来参考：

任务：{task}
结果：{result}
用户反馈：{feedback}

规则格式：当[场景]时，应该[做法]，避免[错误做法]。"""

        rule = (await self.llm.ainvoke(prompt)).content
        self.learned_rules.append(rule)

        # 记录经验
        self.exp_store.record_experience(
            task=task, plan="", result=result,
            success="好" in feedback or "正确" in feedback,
            feedback=feedback,
        )

    def get_enhanced_prompt(self, task: str) -> str:
        """生成增强提示（包含历史经验和规则）"""
        # 检索相关经验
        experiences = self.exp_store.recall_similar(task, k=3)
        exp_text = "\n".join(doc.page_content for doc in experiences)

        # 组合规则
        rules_text = "\n".join(f"- {r}" for r in self.learned_rules[-10:])

        return f"""任务：{task}

参考经验：
{exp_text}

注意事项：
{rules_text}"""
```
