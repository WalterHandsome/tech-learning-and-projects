from langchain.memory import ConversationBufferMemory

## 3. 摘要记忆

```python
from langchain.memory import ConversationSummaryBufferMemory

# 超过 token 限制时自动摘要
summary_memory = ConversationSummaryBufferMemory(
    llm=ChatOpenAI(model="gpt-4o-mini"),
    max_token_limit=2000,
    return_messages=True,
)

# 手动实现摘要记忆
class SummaryMemory:
    def __init__(self, llm, max_messages: int = 20):
        self.llm = llm
        self.max_messages = max_messages
        self.messages: list = []
        self.summary: str = ""

    def add_message(self, role: str, content: str):
        self.messages.append({"role": role, "content": content})
        if len(self.messages) > self.max_messages:
            self._compress()

    def _compress(self):
        """将旧消息压缩为摘要"""
        old_messages = self.messages[:10]
        prompt = f"请将以下对话摘要为简洁的要点：\n{old_messages}"
        new_summary = self.llm.invoke(prompt).content
        self.summary = f"{self.summary}\n{new_summary}"
        self.messages = self.messages[10:]

    def get_context(self) -> str:
        context = f"历史摘要：{self.summary}\n" if self.summary else ""
        context += "\n".join(f"{m['role']}: {m['content']}" for m in self.messages)
        return context
```

## 4. 向量记忆

```python
from langchain_community.vectorstores import Chroma
from langchain_openai import OpenAIEmbeddings

class VectorMemory:
    """基于向量的长期记忆"""

    def __init__(self):
        self.store = Chroma(
            collection_name="agent_memory",
            embedding_function=OpenAIEmbeddings(model="text-embedding-3-small"),
            persist_directory="./memory_db",
        )

    def store_memory(self, content: str, metadata: dict = None):
        """存储记忆"""
        self.store.add_texts(
            texts=[content],
            metadatas=[metadata or {}],
        )

    def recall(self, query: str, k: int = 5) -> list[str]:
        """检索相关记忆"""
        docs = self.store.similarity_search(query, k=k)
        return [doc.page_content for doc in docs]

memory = VectorMemory()
memory.store_memory("用户偏好：喜欢简洁的代码风格", {"type": "preference"})
memory.store_memory("上次讨论了 RAG 架构设计", {"type": "topic"})

# 检索相关记忆
relevant = memory.recall("代码风格建议")
```

## 5. 实体记忆

```python
class EntityMemory:
    """实体记忆：追踪对话中提到的实体信息"""

    def __init__(self, llm):
        self.llm = llm
        self.entities: dict[str, dict] = {}

    async def extract_and_update(self, message: str):
        """从消息中提取实体并更新"""
        prompt = f"""从以下消息中提取实体信息，JSON 格式输出：
消息：{message}
已知实体：{json.dumps(self.entities, ensure_ascii=False)}

输出格式：{{"entity_name": {{"type": "类型", "attributes": {{}}}}}}"""

        result = await self.llm.ainvoke(prompt)
        new_entities = json.loads(result.content)
        # 合并更新
        for name, info in new_entities.items():
            if name in self.entities:
                self.entities[name]["attributes"].update(info.get("attributes", {}))
            else:
                self.entities[name] = info

    def get_entity(self, name: str) -> dict:
        return self.entities.get(name, {})

    def get_context(self) -> str:
        if not self.entities:
            return ""
        return "已知实体：\n" + "\n".join(
            f"- {name}: {info}" for name, info in self.entities.items()
        )

entity_mem = EntityMemory(llm)
await entity_mem.extract_and_update("张三是产品经理，负责电商项目")
await entity_mem.extract_and_update("张三的项目下周要上线")
print(entity_mem.get_entity("张三"))
# {"type": "人物", "attributes": {"role": "产品经理", "project": "电商项目", "deadline": "下周"}}
```

## 6. 记忆检索策略

```python
class MemoryRetriever:
    """综合记忆检索：结合多种记忆源"""

    def __init__(self, summary_mem, vector_mem, entity_mem):
        self.summary = summary_mem
        self.vector = vector_mem
        self.entity = entity_mem

    def retrieve(self, query: str) -> str:
        """综合检索所有记忆源"""
        context_parts = []

        # 1. 实体记忆（精确匹配）
        entity_ctx = self.entity.get_context()
        if entity_ctx:
            context_parts.append(entity_ctx)

        # 2. 向量记忆（语义相关）
        relevant = self.vector.recall(query, k=3)
        if relevant:
            context_parts.append("相关记忆：\n" + "\n".join(relevant))

        # 3. 对话摘要（全局上下文）
        summary = self.summary.get_context()
        if summary:
            context_parts.append(summary)

        return "\n\n".join(context_parts)
```
