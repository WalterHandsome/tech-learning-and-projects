## 1. 多轮对话管理

```python
from dataclasses import dataclass, field
from datetime import datetime

@dataclass
class Message:
    role: str          # system / user / assistant / tool
    content: str
    timestamp: datetime = field(default_factory=datetime.now)
    token_count: int = 0

class ConversationManager:
    """多轮对话管理器"""

    def __init__(self, system_prompt: str, max_tokens: int = 8000):
        self.system_prompt = system_prompt
        self.max_tokens = max_tokens
        self.messages: list[Message] = []

    def add_message(self, role: str, content: str, token_count: int = 0):
        self.messages.append(Message(role=role, content=content, token_count=token_count))

    def get_messages(self) -> list[dict]:
        """获取发送给 LLM 的消息列表"""
        result = [{"role": "system", "content": self.system_prompt}]
        # 从最新消息开始，确保不超过 token 限制
        total_tokens = estimate_tokens(self.system_prompt)
        selected = []
        for msg in reversed(self.messages):
            msg_tokens = msg.token_count or estimate_tokens(msg.content)
            if total_tokens + msg_tokens > self.max_tokens:
                break
            selected.insert(0, {"role": msg.role, "content": msg.content})
            total_tokens += msg_tokens
        return result + selected

def estimate_tokens(text: str) -> int:
    """粗略估算 token 数（中文约 1 字 = 1.5 token）"""
    return int(len(text) * 1.5)
```

## 2. 上下文窗口优化

```python
class ContextOptimizer:
    """上下文窗口优化器"""

    def __init__(self, llm, max_context_tokens: int = 4000):
        self.llm = llm
        self.max_tokens = max_context_tokens

    def truncate_middle(self, messages: list[dict], max_tokens: int) -> list[dict]:
        """中间截断：保留开头和结尾，截断中间"""
        if self._count_tokens(messages) <= max_tokens:
            return messages

        # 保留 system + 最早 2 条 + 最近 N 条
        head = messages[:3]  # system + 前 2 条
        tail = messages[-5:]  # 最近 5 条

        if self._count_tokens(head + tail) <= max_tokens:
            return head + [{"role": "system", "content": "[...中间对话已省略...]"}] + tail
        return head + tail[-3:]

    async def compress_history(self, messages: list[dict]) -> list[dict]:
        """压缩历史：将旧消息摘要化"""
        if self._count_tokens(messages) <= self.max_tokens:
            return messages

        # 将前半部分压缩为摘要
        mid = len(messages) // 2
        old_messages = messages[1:mid]  # 跳过 system
        recent_messages = messages[mid:]

        summary = await self.llm.ainvoke(
            f"请用 3-5 句话摘要以下对话的关键信息：\n{old_messages}"
        )

        return [
            messages[0],  # system prompt
            {"role": "system", "content": f"对话历史摘要：{summary.content}"},
            *recent_messages,
        ]

    def _count_tokens(self, messages: list[dict]) -> int:
        return sum(estimate_tokens(m["content"]) for m in messages)
```

## 3. 对话历史压缩

```python
class ConversationCompressor:
    """对话历史渐进式压缩"""

    def __init__(self, llm):
        self.llm = llm
        self.compressed_summary = ""
        self.recent_messages: list[dict] = []
        self.compress_threshold = 15  # 超过 15 条触发压缩

    async def add_and_compress(self, message: dict):
        self.recent_messages.append(message)

        if len(self.recent_messages) > self.compress_threshold:
            # 压缩前 10 条为摘要
            to_compress = self.recent_messages[:10]
            prompt = f"""当前摘要：{self.compressed_summary}

新对话内容：
{self._format_messages(to_compress)}

请更新摘要，保留所有关键信息（用户偏好、决策、重要数据）："""

            result = await self.llm.ainvoke(prompt)
            self.compressed_summary = result.content
            self.recent_messages = self.recent_messages[10:]

    def get_full_context(self) -> str:
        parts = []
        if self.compressed_summary:
            parts.append(f"[历史摘要] {self.compressed_summary}")
        parts.extend(self._format_messages(self.recent_messages).split("\n"))
        return "\n".join(parts)

    def _format_messages(self, messages: list[dict]) -> str:
        return "\n".join(f"{m['role']}: {m['content']}" for m in messages)
```

## 4. 会话隔离

```python
from typing import Optional
import uuid

class SessionManager:
    """会话管理器：隔离不同用户/会话的上下文"""

    def __init__(self):
        self.sessions: dict[str, ConversationManager] = {}

    def create_session(self, user_id: str, system_prompt: str) -> str:
        """创建新会话"""
        session_id = f"{user_id}_{uuid.uuid4().hex[:8]}"
        self.sessions[session_id] = ConversationManager(
            system_prompt=system_prompt,
            max_tokens=8000,
        )
        return session_id

    def get_session(self, session_id: str) -> Optional[ConversationManager]:
        return self.sessions.get(session_id)

    def cleanup_expired(self, max_age_hours: int = 24):
        """清理过期会话"""
        now = datetime.now()
        expired = [
            sid for sid, conv in self.sessions.items()
            if conv.messages and (now - conv.messages[-1].timestamp).total_seconds() > max_age_hours * 3600
        ]
        for sid in expired:
            del self.sessions[sid]

# 使用
sm = SessionManager()
sid = sm.create_session("user-123", "你是客服助手")
session = sm.get_session(sid)
session.add_message("user", "你好")
session.add_message("assistant", "你好！有什么可以帮助你的？")
```
