## 1. Human-in-the-Loop 模式

```
┌──────────┐    ┌──────────┐    ┌──────────┐
│  Agent    │ →  │ 审批节点  │ →  │ 继续执行  │
│  执行任务  │    │ 人工审核  │    │ 或取消    │
└──────────┘    └──────────┘    └──────────┘

常见触发点：
- 敏感操作（删除、支付、发布）
- 不确定性高的决策
- 涉及外部系统的操作
- 超出 Agent 权限范围
```

## 2. LangGraph 中断与恢复

```python
from langgraph.graph import StateGraph, START, END
from langgraph.types import interrupt, Command
from langgraph.checkpoint.memory import MemorySaver

class WorkflowState(TypedDict):
    task: str
    plan: str
    result: str
    approved: bool

def plan_task(state: WorkflowState) -> dict:
    """Agent 制定执行计划"""
    plan = llm.invoke(f"为以下任务制定执行计划：{state['task']}").content
    return {"plan": plan}

def human_review(state: WorkflowState) -> dict:
    """人工审核节点"""
    # interrupt 会暂停执行，等待人工输入
    decision = interrupt({
        "type": "approval",
        "plan": state["plan"],
        "message": "请审核以上执行计划，输入 approve 或 reject",
    })
    return {"approved": decision == "approve"}

def execute_task(state: WorkflowState) -> dict:
    if not state["approved"]:
        return {"result": "任务已被人工取消"}
    result = llm.invoke(f"执行计划：{state['plan']}").content
    return {"result": result}

graph = StateGraph(WorkflowState)
graph.add_node("plan", plan_task)
graph.add_node("review", human_review)
graph.add_node("execute", execute_task)
graph.add_edge(START, "plan")
graph.add_edge("plan", "review")
graph.add_edge("review", "execute")
graph.add_edge("execute", END)

app = graph.compile(checkpointer=MemorySaver())

# 第一次运行：会在 review 节点暂停
config = {"configurable": {"thread_id": "task-001"}}
result = app.invoke({"task": "删除过期用户数据"}, config)

# 人工审核后恢复执行
result = app.invoke(Command(resume="approve"), config)
```

## 3. 审批工作流

```python
class ApprovalWorkflow:
    """多级审批工作流"""

    def __init__(self):
        self.approval_rules = {
            "low_risk": {"approvers": 0},           # 自动通过
            "medium_risk": {"approvers": 1},         # 1 人审批
            "high_risk": {"approvers": 2},           # 2 人审批
            "critical": {"approvers": 3, "admin": True},  # 3 人 + 管理员
        }

    def classify_risk(self, action: dict) -> str:
        """评估操作风险等级"""
        if action["type"] in ["delete", "payment"]:
            return "high_risk"
        if action["type"] in ["update", "create"]:
            return "medium_risk"
        return "low_risk"

    async def request_approval(self, action: dict, user_id: str) -> bool:
        risk = self.classify_risk(action)
        rule = self.approval_rules[risk]

        if rule["approvers"] == 0:
            return True

        # 发送审批请求
        approval_request = {
            "action": action,
            "risk_level": risk,
            "requested_by": user_id,
            "required_approvals": rule["approvers"],
        }
        # 等待审批结果（实际中通过消息队列/WebSocket）
        return await wait_for_approvals(approval_request)

workflow = ApprovalWorkflow()
approved = await workflow.request_approval(
    {"type": "delete", "target": "expired_users", "count": 1500},
    user_id="agent-001",
)
```

## 4. 信任等级

```python
from enum import IntEnum

class TrustLevel(IntEnum):
    NONE = 0        # 所有操作需审批
    LOW = 1         # 读操作自动，写操作审批
    MEDIUM = 2      # 常规操作自动，敏感操作审批
    HIGH = 3        # 大部分自动，仅关键操作审批
    FULL = 4        # 完全自主（仅限内部系统）

class TrustManager:
    def __init__(self):
        self.agent_trust: dict[str, TrustLevel] = {}
        self.operation_threshold: dict[str, TrustLevel] = {
            "read_data": TrustLevel.LOW,
            "write_data": TrustLevel.MEDIUM,
            "delete_data": TrustLevel.HIGH,
            "system_config": TrustLevel.FULL,
        }

    def needs_approval(self, agent_id: str, operation: str) -> bool:
        agent_level = self.agent_trust.get(agent_id, TrustLevel.NONE)
        required_level = self.operation_threshold.get(operation, TrustLevel.HIGH)
        return agent_level < required_level

trust = TrustManager()
trust.agent_trust["customer-service-bot"] = TrustLevel.MEDIUM
print(trust.needs_approval("customer-service-bot", "read_data"))    # False
print(trust.needs_approval("customer-service-bot", "delete_data"))  # True
```

## 5. 渐进式自主

```python
class ProgressiveAutonomy:
    """渐进式自主：根据 Agent 表现动态调整信任等级"""

    def __init__(self):
        self.success_count: dict[str, int] = {}
        self.error_count: dict[str, int] = {}
        self.thresholds = {
            TrustLevel.LOW: 0,
            TrustLevel.MEDIUM: 10,    # 成功 10 次升级
            TrustLevel.HIGH: 50,      # 成功 50 次升级
        }

    def record_outcome(self, agent_id: str, success: bool):
        if success:
            self.success_count[agent_id] = self.success_count.get(agent_id, 0) + 1
        else:
            self.error_count[agent_id] = self.error_count.get(agent_id, 0) + 1
            # 失败时降级
            self.success_count[agent_id] = max(0, self.success_count.get(agent_id, 0) - 5)

    def get_trust_level(self, agent_id: str) -> TrustLevel:
        successes = self.success_count.get(agent_id, 0)
        for level in reversed(list(TrustLevel)):
            if successes >= self.thresholds.get(level, float("inf")):
                return level
        return TrustLevel.NONE
```
