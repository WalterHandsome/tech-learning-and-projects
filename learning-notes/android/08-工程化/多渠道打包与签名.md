# 多渠道打包与签名

## 1. 签名配置

```kotlin
// app/build.gradle.kts
android {
    signingConfigs {
        create("release") {
            storeFile = file(System.getenv("KEYSTORE_PATH") ?: "keystore/release.jks")
            storePassword = System.getenv("KEYSTORE_PASSWORD") ?: ""
            keyAlias = System.getenv("KEY_ALIAS") ?: ""
            keyPassword = System.getenv("KEY_PASSWORD") ?: ""
        }
    }

    buildTypes {
        release {
            signingConfig = signingConfigs.getByName("release")
        }
    }
}

// local.properties（不提交到 Git）
// KEYSTORE_PATH=keystore/release.jks
// KEYSTORE_PASSWORD=your_password
// KEY_ALIAS=your_alias
// KEY_PASSWORD=your_key_password
```

## 2. productFlavors 多渠道

```kotlin
android {
    flavorDimensions += listOf("channel", "api")

    productFlavors {
        // 渠道维度
        create("google") { dimension = "channel" }
        create("huawei") {
            dimension = "channel"
            applicationIdSuffix = ".huawei"
        }
        create("xiaomi") { dimension = "channel" }

        // API 维度
        create("free") {
            dimension = "api"
            buildConfigField("Boolean", "IS_PREMIUM", "false")
        }
        create("premium") {
            dimension = "api"
            buildConfigField("Boolean", "IS_PREMIUM", "true")
        }
    }

    // 过滤不需要的变体
    androidComponents {
        beforeVariants { variant ->
            if (variant.flavorName == "xiaomiPremium") {
                variant.enable = false
            }
        }
    }
}

// 渠道特定资源
// app/src/google/res/values/strings.xml
// app/src/huawei/res/values/strings.xml
// app/src/google/kotlin/com/example/PushService.kt
```

## 3. V1 / V2 / V3 签名

```kotlin
android {
    signingConfigs {
        create("release") {
            // ...
            enableV1Signing = true   // JAR 签名（兼容旧设备）
            enableV2Signing = true   // APK 签名 v2（Android 7.0+）
            enableV3Signing = true   // APK 签名 v3（Android 9.0+，支持密钥轮换）
            enableV4Signing = true   // APK 签名 v4（Android 11+，增量安装）
        }
    }
}

// 验证签名
// apksigner verify --verbose app-release.apk
// keytool -printcert -jarfile app-release.apk
```

## 5. 自动化版本号

```kotlin
// build.gradle.kts
val versionMajor = 1
val versionMinor = 2
val versionPatch = 3

android {
    defaultConfig {
        versionCode = versionMajor * 10000 + versionMinor * 100 + versionPatch
        versionName = "$versionMajor.$versionMinor.$versionPatch"
    }
}

// 基于 Git 的版本号
fun gitVersionCode(): Int {
    val process = Runtime.getRuntime().exec("git rev-list --count HEAD")
    return process.inputStream.bufferedReader().readText().trim().toIntOrNull() ?: 1
}

fun gitVersionName(): String {
    val process = Runtime.getRuntime().exec("git describe --tags --abbrev=0")
    return process.inputStream.bufferedReader().readText().trim().ifEmpty { "0.0.1" }
}
```
