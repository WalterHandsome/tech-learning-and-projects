## 1. CameraX

```kotlin
// 依赖: androidx.camera:camera-camera2, camera-lifecycle, camera-view

@Composable
fun CameraPreview(onImageCaptured: (Uri) -> Unit) {
    val context = LocalContext.current
    val lifecycleOwner = LocalLifecycleOwner.current
    val previewView = remember { PreviewView(context) }

    LaunchedEffect(Unit) {
        val cameraProvider = ProcessCameraProvider.getInstance(context).await()
        val preview = Preview.Builder().build().also {
            it.setSurfaceProvider(previewView.surfaceProvider)
        }
        val imageCapture = ImageCapture.Builder()
            .setCaptureMode(ImageCapture.CAPTURE_MODE_MINIMIZE_LATENCY)
            .build()

        cameraProvider.unbindAll()
        cameraProvider.bindToLifecycle(
            lifecycleOwner, CameraSelector.DEFAULT_BACK_CAMERA, preview, imageCapture
        )
    }

    AndroidView(factory = { previewView }, modifier = Modifier.fillMaxSize())
}

// 拍照
fun takePhoto(imageCapture: ImageCapture, context: Context, onSaved: (Uri) -> Unit) {
    val photoFile = File(context.cacheDir, "photo_${System.currentTimeMillis()}.jpg")
    val outputOptions = ImageCapture.OutputFileOptions.Builder(photoFile).build()

    imageCapture.takePicture(outputOptions, ContextCompat.getMainExecutor(context),
        object : ImageCapture.OnImageSavedCallback {
            override fun onImageSaved(output: ImageCapture.OutputFileResults) {
                onSaved(Uri.fromFile(photoFile))
            }
            override fun onError(exception: ImageCaptureException) {
                Log.e("Camera", "拍照失败", exception)
            }
        }
    )
}
```

## 3. ExoPlayer（Media3）

```kotlin
// 依赖: androidx.media3:media3-exoplayer, media3-ui

@Composable
fun VideoPlayer(videoUrl: String) {
    val context = LocalContext.current
    val player = remember {
        ExoPlayer.Builder(context).build().apply {
            setMediaItem(MediaItem.fromUri(videoUrl))
            prepare()
        }
    }

    DisposableEffect(Unit) {
        onDispose { player.release() }
    }

    AndroidView(
        factory = { ctx ->
            PlayerView(ctx).apply { this.player = player }
        },
        modifier = Modifier.fillMaxWidth().aspectRatio(16f / 9f)
    )
}

// 播放列表
val playlist = listOf(
    MediaItem.fromUri("https://example.com/video1.mp4"),
    MediaItem.fromUri("https://example.com/video2.mp4")
)
player.setMediaItems(playlist)
player.prepare()
player.play()
```

## 4. Coil 图片加载

```kotlin
// 依赖: io.coil-kt:coil-compose

@Composable
fun UserAvatar(imageUrl: String, size: Dp = 48.dp) {
    AsyncImage(
        model = ImageRequest.Builder(LocalContext.current)
            .data(imageUrl)
            .crossfade(true)
            .placeholder(R.drawable.ic_avatar_placeholder)
            .error(R.drawable.ic_avatar_error)
            .transformations(CircleCropTransformation())
            .build(),
        contentDescription = "头像",
        modifier = Modifier.size(size).clip(CircleShape),
        contentScale = ContentScale.Crop
    )
}

// 预加载
val imageLoader = ImageLoader.Builder(context)

            .directory(context.cacheDir.resolve("image_cache"))
            .maxSizePercent(0.02)
            .build()
    }
    .build()
```

## 6. 2026 版本演进

### 6.1 CameraX 1.5/1.6：Compose 原生支持

CameraX 1.5/1.6 引入了 **CameraXViewfinder**，彻底告别 `AndroidView(PreviewView)` 的 "View 孤岛" 模式。

**核心变化**：
- `CameraXViewfinder` 直接消费 `SurfaceRequest`，无需 `AndroidView` 和 `PreviewView`
- 原生 `ContentScale` 和 `Alignment` 替代 `PreviewView.ScaleType`
- EMBEDDED 模式支持完整 `graphicsLayer`（模糊、透明度、变换）
- 响应式坐标变换，手势映射直接在 Compose 中完成

```kotlin
// 依赖: androidx.camera:camera-compose:1.5.0+

@Composable
fun CameraPreview(
    surfaceRequest: SurfaceRequest,
    modifier: Modifier = Modifier
) {
    // 纯 Compose 实现，无需 AndroidView
    CameraXViewfinder(
        surfaceRequest = surfaceRequest,
        modifier = modifier,
        // 使用 Compose 原生的缩放和对齐
        contentScale = ContentScale.Fit,
        alignment = Alignment.Center,
        // EMBEDDED 模式支持 graphicsLayer 效果
        implementationMode = ImplementationMode.EMBEDDED
    )
}
```

**CameraX 1.5 其他新特性**：
- 专业级视频录制能力增强
- 更灵活的相机会话配置 API
- 改进的设备兼容性

> 来源：[Compose-Native CameraX in 2026](https://proandroiddev.com/compose-native-camerax-in-2026-the-complete-guide-bf36c76a78e9)、[CameraX 1.5 发布公告](https://android-developers.googleblog.com/2025/11/introducing-camerax-15-powerful-video.html)

### 6.2 Media3 1.10：Material3 播放组件

Media3 1.10（2026-03）引入了 Compose Material3 播放组件，告别 `AndroidView(PlayerView)` 模式。

**核心新特性**：
- **Player Composable**：开箱即用的 Material3 播放器组件（ContentFrame + 自定义控件）
- **ProgressSlider**：拖拽和点击手势的进度条 Composable
- **PlaybackSpeedControl / PlaybackSpeedToggleButton**：播放速度管理组件
- **格式支持扩展**：Dolby Vision Profile 10、VVC（H.266）MP4 提取
- **IAMF 双耳输出**：通过 iamf_tools 或 Android OS Spatializer 无缝支持
- **实验性动态调度**：`experimentalSetDynamicSchedulingEnabled()` 优化播放循环效率
- **新模块拆分**：FrameExtractor → `media3-inspector-frame`，Lottie → `media3-effect-lottie`

```kotlin
// 依赖: androidx.media3:media3-ui-compose-material3:1.10.0

@Composable
fun VideoPlayer(videoUrl: String) {
    val context = LocalContext.current
    val player = remember {
        ExoPlayer.Builder(context)
            // 实验性：更高效的播放循环调度
            .experimentalSetDynamicSchedulingEnabled(true)
            .build()
            .apply {
                setMediaItem(MediaItem.fromUri(videoUrl))
                prepare()
            }
    }

    DisposableEffect(Unit) {
        onDispose { player.release() }
    }

    // Material3 风格的播放器组件（纯 Compose）
    Player(
        player = player,
        modifier = Modifier.fillMaxWidth().aspectRatio(16f / 9f)
    )
}
```

> 来源：[Media3 1.10 is out](https://developer.android.com/blog/posts/media3-1-10-is-out)

### 6.3 Coil 3.x：Compose Multiplatform 图片加载

Coil 3.x 是完全重写的版本，支持 Compose Multiplatform。

**核心变化**：
- **Maven 坐标变更**：`io.coil-kt` → `io.coil-kt.coil3`，包名 `coil` → `coil3`
- **网络库可选**：`coil-core` 不再默认包含网络支持，需显式添加 `coil-network-okhttp` 或 `coil-network-ktor3`
- **KMP 支持**：Android、iOS、Desktop、Web 全平台
- **ConcurrentRequestStrategy**（3.4.0）：合并相同 key 的并发网络请求

```kotlin
// 依赖: io.coil-kt.coil3:coil-compose:3.4.0
// 网络: io.coil-kt.coil3:coil-network-okhttp:3.4.0

// Coil 3.x 用法（包名从 coil 改为 coil3）
import coil3.compose.AsyncImage
import coil3.request.ImageRequest
import coil3.transform.CircleCropTransformation

@Composable
fun UserAvatar(imageUrl: String, size: Dp = 48.dp) {
    AsyncImage(
        model = ImageRequest.Builder(LocalContext.current)
            .data(imageUrl)
            .crossfade(true)
            .transformations(CircleCropTransformation())
            .build(),
        contentDescription = "头像",
        modifier = Modifier.size(size).clip(CircleShape),
        contentScale = ContentScale.Crop
    )
}
```

### 6.4 2026 年多媒体选型建议

| 场景 | 推荐方案 | 说明 |
|------|---------|------|
| 相机预览（Compose） | CameraXViewfinder（1.5+） | 纯 Compose，无需 AndroidView |
| 相机预览（View） | PreviewView | 仍然稳定可用 |
| 视频播放（Compose） | Media3 Player Composable（1.10+） | Material3 风格，纯 Compose |
| 视频播放（View） | Media3 PlayerView | 传统 View 方式 |
| 图片加载（新项目） | Coil 3.x | KMP 支持，Compose 优先 |
| 图片加载（维护项目） | Coil 2.x / Glide | 稳定，无需迁移 |
