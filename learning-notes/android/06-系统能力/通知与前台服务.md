## 1. NotificationChannel（Android 8.0+）

```kotlin
class MyApp : Application() {
    override fun onCreate() {
        super.onCreate()
        createNotificationChannels()
    }

    private fun createNotificationChannels() {
        val channels = listOf(
            NotificationChannel("messages", "消息通知", NotificationManager.IMPORTANCE_HIGH).apply {
                description = "聊天消息通知"
                enableVibration(true)
                enableLights(true)
            },
            NotificationChannel("updates", "更新通知", NotificationManager.IMPORTANCE_DEFAULT).apply {
                description = "应用更新提醒"
            },
            NotificationChannel("background", "后台任务", NotificationManager.IMPORTANCE_LOW).apply {
                description = "后台任务进度"
            }
        )
        val manager = getSystemService(NotificationManager::class.java)
        manager.createNotificationChannels(channels)
    }
}
```

## 2. 发送通知

```kotlin
class NotificationHelper(private val context: Context) {

    // 基本通知
    fun showSimpleNotification(title: String, content: String) {
        val intent = Intent(context, MainActivity::class.java)
        val pendingIntent = PendingIntent.getActivity(
            context, 0, intent, PendingIntent.FLAG_IMMUTABLE
        )

        val notification = NotificationCompat.Builder(context, "messages")
            .setSmallIcon(R.drawable.ic_notification)
            .setContentTitle(title)
            .setContentText(content)
            .setPriority(NotificationCompat.PRIORITY_HIGH)
            .setContentIntent(pendingIntent)
            .setAutoCancel(true)
            .build()

        NotificationManagerCompat.from(context).notify(1, notification)
    }

    // 大文本通知
    fun showBigTextNotification(title: String, longText: String) {
        val notification = NotificationCompat.Builder(context, "messages")
            .setSmallIcon(R.drawable.ic_notification)
            .setContentTitle(title)
            .setStyle(NotificationCompat.BigTextStyle().bigText(longText))
            .build()

        NotificationManagerCompat.from(context).notify(2, notification)
    }

    // 进度通知
    fun showProgressNotification(title: String, progress: Int) {
        val notification = NotificationCompat.Builder(context, "background")
            .setSmallIcon(R.drawable.ic_download)
            .setContentTitle(title)
            .setProgress(100, progress, false)
            .setOngoing(true)
            .build()

        NotificationManagerCompat.from(context).notify(3, notification)
    }

    // 带操作按钮的通知
    fun showActionNotification(title: String, content: String) {
        val replyIntent = PendingIntent.getBroadcast(
            context, 0, Intent("REPLY_ACTION"), PendingIntent.FLAG_IMMUTABLE
        )
        val notification = NotificationCompat.Builder(context, "messages")
            .setSmallIcon(R.drawable.ic_notification)
            .setContentTitle(title)
            .setContentText(content)
            .addAction(R.drawable.ic_reply, "回复", replyIntent)
            .addAction(R.drawable.ic_dismiss, "忽略", null)
            .build()

        NotificationManagerCompat.from(context).notify(4, notification)
    }
}
```

## 3. 前台服务通知

```kotlin
class UploadForegroundService : Service() {
    override fun onStartCommand(intent: Intent?, flags: Int, startId: Int): Int {
        val notification = createNotification("准备上传...", 0)

        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.UPSIDE_DOWN_CAKE) {
            startForeground(100, notification, ServiceInfo.FOREGROUND_SERVICE_TYPE_DATA_SYNC)
        } else {
            startForeground(100, notification)
        }

        startUpload()
        return START_NOT_STICKY
    }

    private fun createNotification(text: String, progress: Int): Notification {
        return NotificationCompat.Builder(this, "background")
            .setSmallIcon(R.drawable.ic_upload)
            .setContentTitle("文件上传")
            .setContentText(text)
            .setProgress(100, progress, progress == 0)
            .setOngoing(true)
            .build()
    }

    private fun updateProgress(progress: Int) {
        val notification = createNotification("上传中 $progress%", progress)
        NotificationManagerCompat.from(this).notify(100, notification)
    }

    override fun onBind(intent: Intent?): IBinder? = null
}
```

## 4. 通知权限（Android 13+）

```kotlin
@Composable
fun NotificationPermissionRequest() {
    if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
        val permissionState = rememberPermissionState(
            Manifest.permission.POST_NOTIFICATIONS
        )
        LaunchedEffect(Unit) {
            if (!permissionState.status.isGranted) {
                permissionState.launchPermissionRequest()
            }
        }
    }
}
```
